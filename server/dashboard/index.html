API Endpoints

GET/api/healthHealth check
GET/api/gamesList games
POST/api/gamesRegister game
GET/api/games/{id}Game details
DEL/api/games/{id}Remove game
POST/api/games/{id}/heartbeatKeep alive

Diagnostic Commands 

# Check server health
curl http://localhost:8080/api/health

# List active games
curl -H "X-API-Key: YOUR_KEY" \
  http://localhost:8080/api/games

# Register a test game
curl -X POST \
  -H "Content-Type: application/json" \
  -H "X-API-Key: YOUR_KEY" \
  -d '{"name":"Test","max_players":4}' \
  http://localhost:8080/api/games

# Test STUN connectivity
stunclient your-server-ip 3478

# View server logs
docker compose logs -f server

# View TURN relay logs
docker compose logs -f coturn
